import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

from sentence_transformers import SentenceTransformer
from sklearn.metrics.pairwise import cosine_similarity






Cell 2 — Create 10 sentences (exactly 3 topics)
sentences = [
    # Cricket (4)
    "The opener scored a century with elegant cover drives.",
    "The bowler swung the ball late and hit the top of off stump.",
    "A tight fielding effort saved ten runs in the final over.",
    "The captain set an attacking field with slips and a short leg.",

    # Cooking (3)
    "Sauté the onions until golden before adding the tomatoes and spices.",
    "Rest the dough for thirty minutes to improve the bread's texture.",
    "Simmer the curry slowly so the flavors deepen and the sauce thickens.",

    # Cybersecurity (3)
    "Multi-factor authentication reduces the risk of account takeover attacks.",
    "The phishing email used a fake login page to steal user credentials.",
    "Regular patching closes vulnerabilities that ransomware often exploits."
]

topics = (["Cricket"]*4) + (["Cooking"]*3) + (["Cybersecurity"]*3)

len(sentences), len(set(topics))





Cell 3 — Generate embeddings (all-MiniLM-L6-v2)

model = SentenceTransformer("all-MiniLM-L6-v2")
embeddings = model.encode(sentences, normalize_embeddings=True)  # normalize helps cosine similarity
embeddings.shape


Cell 4 — 10×10 cosine similarity matrix + heatmap

sim = cosine_similarity(embeddings)

df_sim = pd.DataFrame(sim, index=[f"{i+1}" for i in range(10)], columns=[f"{i+1}" for i in range(10)])

plt.figure(figsize=(10, 8))
sns.heatmap(df_sim, annot=True, fmt=".2f", cmap="YlGnBu", square=True)
plt.title("Cosine Similarity (10×10) — Sentence Embeddings")
plt.xlabel("Sentence #")
plt.ylabel("Sentence #")
plt.show()



Cell 5 — Query: top 2 most similar sentences + scores


query = "The bowler took three wickets in one over"
q_emb = model.encode([query], normalize_embeddings=True)

scores = cosine_similarity(q_emb, embeddings).flatten()
top2_idx = scores.argsort()[::-1][:2]

print("Query:", query, "\n")
for rank, idx in enumerate(top2_idx, start=1):
    print(f"Top {rank}: (Sentence #{idx+1}) [{topics[idx]}] {sentences[idx]}")
    print(f"Similarity: {scores[idx]:.4f}\n")
